int[] a = {2, 1, 3};
Arrays.sort(a); //{1, 2, 3}
Arrays.sort(a, 0, 2); //对下标0-2排序
Arrays.sort(a, (a, b) -> {
  return b - a; //返回值大于0交换
}); //{3, 2, 1}

//java中"\\"表示"\"
A       //指定字符
.       //任意字符
\\d     //任意数字，等价[0-9]
\\w     //任意数字大小写字母下划线，等价[0-9a-zA-Z_]
\\s     //空格，tab
\\D     //非\\d
\\W     //非\\w
\\S     //非\\s
A*      //任意个数
A+      //至少一个
A?      //零或一个
A{3}    //指定个数
A{2,3}  //指定范围个数
A{2,}   //最少n个
A{0,3}  //最多n个
^A      //匹配开头
A$      //匹配结尾
[ABC]   //[]内任意字符
[^ABC]  //非[]内任意字符
A|B|C   //或
?       //非贪婪匹配
